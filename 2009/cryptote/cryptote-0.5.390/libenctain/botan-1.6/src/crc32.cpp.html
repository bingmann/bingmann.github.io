/*************************************************
* Finalize a CRC32 Checksum                      *
*************************************************/
void CRC32::final_result(byte output[])
   {
   crc ^= 0xFFFFFFFF;
   for(u32bit j = 0; j != 4; ++j)
      output[j] = get_byte(j, crc);
   clear();
   }
