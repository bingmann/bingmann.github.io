00056 template <typename T>
00057 void matrix_multiply(const matrix2d<T> & a, const matrix2d<T> & b, matrix2d<T> & c)
00058 {
00059     assert(a.get_width() == b.get_height());
00060     assert(b.get_height() == c.get_height());
00061     assert(b.get_width() == c.get_width());
00062 
00063     for (stxxl::unsigned_type y = 0; y < c.get_height(); ++y)
00064         for (stxxl::unsigned_type x = 0; x < c.get_width(); ++x)
00065         {
00066             c.element(x, y) = 0;
00067             for (stxxl::unsigned_type t = 0; t < a.get_width(); ++t)
00068                 c.element(x, y) += a.const_element(t, y) * b.const_element(x, t);
00069         }
00070 }
