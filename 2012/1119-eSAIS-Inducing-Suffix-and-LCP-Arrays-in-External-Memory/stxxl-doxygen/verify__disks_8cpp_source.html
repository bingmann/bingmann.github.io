00055 void watch_times(request_ptr reqs[], unsigned n, double * out)
00056 {
00057     bool * finished = new bool[n];
00058     unsigned count = 0;
00059     for (unsigned i = 0; i < n; i++)
00060         finished[i] = false;
00061

00063     {
00064         usleep(POLL_DELAY);

00068             if (!finished[i])
