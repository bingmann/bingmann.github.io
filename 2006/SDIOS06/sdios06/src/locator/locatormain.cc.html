int main()
{
	LogMessage( "[locator] Locator is alive..." );
	locator = new Locator();

	locator->Register(ROOT_DIRECTORY_HANDLE,
					  "task",
					  static_cast<interfaceid_t>(IF_DIRECTORY_ID),
					  L4_Pager(),
					  static_cast<objectid_t>(0) );
	
	locator_server();
	
	panic( "Reached end of locator program..\n" );	
	return( 1 );
}
