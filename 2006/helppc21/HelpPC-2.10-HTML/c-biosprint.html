               TC: int biosprint( int cmd, int byte, int port )

       - prototype in bios.h

       - outputs byte to port
       - port = 0 (LPT1)
                1 (LPT2), etc...
       - cmd = 0 (print byte);
               1 (init printer port);
               2 (get LPT status)

       - returns status of (values are or'd flags):
         0x01 time out (cmd=0 means output error)
         0x08 I/O error
         0x10 selected
         0x20 out of paper
         0x40 acknowledge
         0x80 not busy

       - MS C uses _bios_printer()
       - see   INT 17
