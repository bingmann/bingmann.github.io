                   TC: void getdate( struct date *dateblk )

       - prototype in dos.h

       - gets DOS date, filling it into the following structures:

       struct date
           {
           int  da_year;               - Year including century
           char da_day;                - Day of the month
           char da_mon;                - Month (1 = Jan)
           };

       - MS C uses _dos_getdate()
       - see  INT 21,2A
