         MSC: unsigned _dos_getfileattr( char *path, unsigned *attr )


       - prototype in dos.h

       - path = any valid DOS file name
       - attr = file attribute:

               _A_NORMAL
               _A_RDONLY
               _A_HIDDEN
               _A_SYSTEM
               _A_VOLID
               _A_SUBDIR
               _A_ARCH

       - returns zero on success; nonzero on failure
       - no corresponding function in Turbo C
       - see   INT 21,43
